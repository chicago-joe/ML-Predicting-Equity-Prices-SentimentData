# INSTALLATION TO GET WALK-FORWARD MODEL WORKING WITH CURRENT DATASET

PYTHON: 3.7.6

1. Create a venv (if not already):
py -3.7 -m venv mlTradingEnv

2. Activate venv:
cd mlTradingEnv\Scripts
activate.bat

3. Update package managers:
python -m pip install -U pip setuptools wheel

4. Install requirements:
python -m pip install -r requirements.txt
